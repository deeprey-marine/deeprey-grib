macro(late_init)
  # Exécute des initialisations après la création du plugin

      ${CMAKE_CURRENT_LIST_DIR}/include/garminxhd
      ${CMAKE_CURRENT_LIST_DIR}/include/navico
      ${CMAKE_CURRENT_LIST_DIR}/include/raymarine

endmacro()
